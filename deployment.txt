 ============================= deployment ============================= 

deployment.yaml

apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deployment
  labels:
    app: nginx
spec:
  replicas: 2
  selector:
    matchLabels:
      app: nginx
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:1.14.2
        ports:
        - containerPort: 80



kubetcl get pods  											->  list all pods
kubectl scale deployment nginx-deployment --replicas=2      ->  without making any changes in yaml file we can scale up and down
kubectl get all 					    					->  list everything which is running on k8s server
kubectl describe deployment nginx-deployment                ->  shows all the logs related to the deployment
kubetcl delete deployment nginx-deployment 				    ->  delete deployment
